238  LOG_DEBUG(m_logger, "Received reply: {}", reply_str);
239 
240  // parse the reply
241  reply_json = json::parse(reply_str);
242  if (reply_json["status"] != "success") {
243  LOG_ERROR(m_logger, "Lookup failed: {}", to_string(reply_json["error"]));
244  throw std::runtime_error("Failed to lookup topic");

246  found = reply_json["found"];

257  new_subscriber->set(zmq::sockopt::rcvhwm, 10);
258  new_subscriber->connect("tcp://" + ip + ":" + port);
259  new_subscriber->set(zmq::sockopt::subscribe, topic.c_str());

483  void start_frame_drop(zmq::socket_t& sub_socket) {
484  // Drop frames that arrive faster than 3ms apart
485  auto timeout = std::chrono::milliseconds(3);
486 
487  // Drop initial burst of frames by using a short poll timeout
488  LOG_INFO(m_logger, "Starting frame drop phase...");
489 
490  zmq::pollitem_t items[] = {
491  { sub_socket, 0, ZMQ_POLLIN, 0 },
492  };
493 
494  while (!m_atomic_stop.load(std::memory_order_relaxed)) {
495  // Poll with timeout
496  int rc = zmq::poll(items, 1, timeout);
497 
498  if (rc == 0) {
499  // Timeout occurred - frames are now arriving slower than our threshold
500  LOG_INFO(m_logger, "Frame rate normalized, continuing normal operation");
501  break;
502  }
503 
504  LOG_DEBUG(m_logger, "Waiting for image...");
505  zmq::message_t topic_msg;
506  zmq::recv_result_t res;
507  try {
508  if (!sub_socket.recv(topic_msg, zmq::recv_flags::none)) {
509  LOG_ERROR(m_logger, "Failed to receive topic");
510  continue;
511  }
512  std::string topic(static_cast<char*>(topic_msg.data()), topic_msg.size());
513  LOG_DEBUG(m_logger, "Received image for topic {}", topic);
514 
515  // Receive the metadata
516  zmq::message_t metadata_msg;
517  if (!sub_socket.recv(metadata_msg, zmq::recv_flags::none)) {
518  LOG_ERROR(m_logger, "Failed to receive metadata");
519  continue;
520  }
521 
522  zmq::message_t image_msg;
523  if (!sub_socket.recv(image_msg, zmq::recv_flags::none)) {
524  LOG_ERROR(m_logger, "Failed to receive image");
525  continue;
526  }
527  } catch (const zmq::error_t& e) {
528  LOG_ERROR(m_logger, "ZMQ error: {}", e.what());
529  continue;
530  }
531  LOG_DEBUG(m_logger, "Dropped frame!");
532  }
533 
534  LOG_INFO(m_logger, "Frame drop phase complete");
535  }
